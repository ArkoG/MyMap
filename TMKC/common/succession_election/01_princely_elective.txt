# The entry should be named in the same way that the elective succession law is named to define it's voter's logic.
# Root scope - Elector, the character voting in the election.
# scope:candidate scope - Candidate, Character considered for the title.
# scope:title scope - Title, The title the nomination is for.
# scope:holder scope - Ruler, The ruler currently holding the title.
# scope:holder_candidate scope - Favored Candidate, The title owner's candidate.
# The voter will vote for the top score candidate calculated from this script.

princely_elective = { #HRE Elective.
	elector_vote_strength = {
		base = 0
		modifier = { #Imperial Dignity
			desc = tooltip_feudal_vote_strength_emperor_tier
			highest_held_title_tier = 5
			add = 2 #ex 2
		}
		modifier = { #Imperial Authority
			desc = imperial_authority
			highest_held_title_tier = 5
			OR = {
				has_realm_law = crown_authority_1
				has_realm_law = crown_authority_2
				has_realm_law = crown_authority_3
			}
			add = {
				value = 0
				
				if = {
					limit = { has_realm_law = crown_authority_1 }
					desc = imperial_CA1
					add = 1
				}
				if = {
					limit = { has_realm_law = crown_authority_2 }
					desc = imperial_CA2
					add = 1
				}
				if = {
					limit = { has_realm_law = crown_authority_3 }
					desc = imperial_CA3
					add = 1
				}
			}
		}

		#modifier = { #Royal Dignity
		#	desc = tooltip_feudal_vote_strength_king_tier
		#	add = 1
		#	highest_held_title_tier = 4
		#}
		modifier = { #Powerful Vassal
			desc = tooltip_feudal_vote_strength_powerful_vassal
			add = {
				value = 2
				if = { #Additional point if the vassal is more powerful than the current ruler (Vassal King who meets this condition will have more voting power than the Emperor himself (1+1+1+2 = 5))
					limit = {
						max_military_strength > scope:holder.max_military_strength
					}
					add = 1
				}
			}
			NOT = { this = scope:holder }
			is_powerful_vassal = yes
			liege = scope:holder
		}
		
	######################## à voir selon ethos. et doctrine ?
		modifier = { #Clerical Influence
			desc = tooltip_feudal_vote_strength_theocrat
			add = 2 #ex 2
			NOT = { this = scope:holder }
			OR = {
				has_government = theocracy_government
				has_council_position = councillor_court_chaplain
			}
			faith = scope:holder.faith
			NOT = { faith = { has_doctrine = doctrine_theocracy_lay_clergy }}
			NOT = { root = scope:holder.faith.religious_head }
			#TODO_CD:Possible trigger checking for investiture succession?
		}
		modifier = { #Clerical Influence
			desc = tooltip_feudal_vote_strength_theocrat
			add = 1 #ex 2
			NOT = { this = scope:holder }
			OR = {
				has_government = theocracy_government
				has_council_position = councillor_court_chaplain
			}
			faith = scope:holder.faith
			faith = { has_doctrine = doctrine_theocracy_lay_clergy }
			NOT = { root = scope:holder.faith.religious_head }
			#TODO_CD:Possible trigger checking for investiture succession?
		}
		modifier = { #Papal Influence
			desc = tooltip_feudal_vote_strength_pope
			add = 5 #If the Pope is an elector, his vote is the most important of all.
			NOT = { this = scope:holder }
			root = scope:holder.faith.religious_head
			#TODO_CD:Possible trigger checking for investiture succession?
		}
		##VICARY
		modifier = { 
			desc = imperial_vicary
			add = 1 
			NOT = { this = scope:holder }
			has_court_position = hre_vicary_court_position
		}
		#Vicary + #Imperial influence
		modifier = { 
			desc = imperial_influence
			NOT = { this = scope:holder }
			has_court_position = hre_vicary_court_position
			has_realm_law = crown_authority_2
			add = {
				value = 1
				if = {
					limit = { has_realm_law = crown_authority_3 }
					desc = imperial_CA3
					add = 1
				}
			}
		}		
		###all others
		modifier = { 
			desc = simple_elector
			add = 1
			NOT = { 
				highest_held_title_tier = 5
				this = scope:holder
				has_court_position = hre_vicary_court_position
				is_powerful_vassal = yes

				has_government = theocracy_government
				has_council_position = councillor_court_chaplain
				
				root = scope:holder.faith.religious_head
			}
		}		
	}
	
	electors = { #character scope
		max = 20  #Seven Prince-Electors + 1 Emperor.
		add = holder #The Emperor
		add = {
			type = holder_direct_vassals
			limit = { 
				#is_vassal_of = scope:holder
				highest_held_title_tier >= tier_county
				highest_held_title_tier < tier_kingdom
				has_court_position = hre_vicary_court_position
			}
		}
		add = {
			type = title_dejure_vassals
			limit = {
				is_vassal_of = scope:holder
				has_government = theocracy_government
				trigger_if = {
					limit = { is_female = yes }
					faith_dominant_gender_female_or_equal = yes
				}
				trigger_else = { faith_dominant_gender_male_or_equal = yes }
			}
		}
		add = {
			#direct dejure vassals
			type = title_dejure_vassals
			limit = {
				#target title of the election
				scope:title = { #Either title is higher than Duke-tier, or must be primary.
					OR = {
						tier > tier_kingdom #tier >= tier_kingdom
						holder = { has_primary_title = prev }
					}
				}
				
				#electors
				#duke or lesser
				
				#no tribal
				NOT = { has_government = tribal_government }
				#tier <= tier_duchy
				
				OR = { #Either de facto direct vassal, or independent ruler who is de jure vassal. (no sub-vassals)
					is_vassal_of = scope:holder
					is_independent_ruler = yes
				}
				trigger_if = { #Must be true priests
					limit = { has_government = theocracy_government }
					trigger_if = {
						limit = { is_female = yes }
						faith_dominant_gender_female_or_equal = yes
					}
					trigger_else = { faith_dominant_gender_male_or_equal = yes }
				}
			}
		}
		add = { #baron tier cities of the emperor's demesne 
			#direct dejure vassals
			type = holder_direct_vassals
			limit = {
				#highest_held_title_tier = tier_county
				has_government = republic_government
			}
		}
		
		priority = {
			base = 1
			modifier = { # The Emperor is always picked.
				add = 200
				this = scope:holder
			}
			modifier = { # designated elector
				add = {
					value = 0
					if = {
						limit = { has_court_position = hre_vicary_court_position }
						add = 100
					}
				}
			}
		
			#PRESTIGIOUS VASSALS
			modifier = { # Prince-Electors. (specific titles, held by rulers of the same faith as the current holder)
				add = 5 #ex 75
				faith = scope:holder.faith
				prestige_level >= 1
				#any_held_title = {
				#	is_target_in_global_variable_list = { name = hre_elector_list target = this }
				#}
			}
			modifier = { # Prince-Electors. (specific titles, held by rulers of the same faith as the current holder)
				add = 5
				faith = scope:holder.faith
				prestige_level >= 2
			}
			modifier = { # Prince-Electors. (specific titles, held by rulers of the same faith as the current holder)
				add = 10
				faith = scope:holder.faith
				prestige_level >= 3
			}
			modifier = { # Prince-Electors. (specific titles, held by rulers of the same faith as the current holder)
				add = 10
				faith = scope:holder.faith
				prestige_level >= 5
			}
			#PIOUS VASSALS
			modifier = { # Prince-Electors. (specific titles, held by rulers of the same faith as the current holder)
				add = 5 #ex 75
				faith = scope:holder.faith
				piety_level >= 1
				#any_held_title = {
				#	is_target_in_global_variable_list = { name = hre_elector_list target = this }
				#}
			}
			modifier = { # Prince-Electors. (specific titles, held by rulers of the same faith as the current holder)
				add = 10
				faith = scope:holder.faith
				piety_level >= 2
			}
			modifier = { # Prince-Electors. (specific titles, held by rulers of the same faith as the current holder)
				add = 10
				faith = scope:holder.faith
				piety_level >= 3
			}
			modifier = { # Prince-Electors. (specific titles, held by rulers of the same faith as the current holder)
				add = 10
				faith = scope:holder.faith
				piety_level >= 5
			}
			#PRESTIGIOUS DYNASTIES
			modifier = { 
				add = -5
				faith = scope:holder.faith
				dynasty = { dynasty_prestige_level <= 1 }
			}
			modifier = { # Prince-Electors. (specific titles, held by rulers of the same faith as the current holder)
				faith = scope:holder.faith
				add = {
					value = 0 
					if = {
						limit = { dynasty = { dynasty_prestige_level >= 2 }}
						add = 5
					}									
					if = {
						limit = { dynasty = { dynasty_prestige_level >= 4 }}
						add = 5
					}
					if = {
						limit = { dynasty = { dynasty_prestige_level >= 6 }}
						add = 10
					}
					if = {
						limit = { dynasty = { dynasty_prestige_level >= 8 }}
						add = 10
					}
				}
			}
			#POWERFUL VASSALS
			modifier = { # Prince-Electors. (specific titles, held by rulers of the same faith as the current holder)
				add = 25
				faith = scope:holder.faith
				is_powerful_vassal = yes
			}
			#THEOCRACIES
			modifier = {
				add = 5
				faith = scope:holder.faith
				has_government = theocracy_government
			}
			modifier = {
				add = 5
				faith = scope:holder.faith
				has_government = theocracy_government
				is_vassal_of = scope:holder
			}
			modifier = {
				add = 5
				faith = scope:holder.faith
				has_government = theocracy_government
				tier > tier_county 
			}			
			#GENERAL
			modifier = { #Fallback Electors
				add = {
					value = 10
					if = { #Slightly favor same faith first.
						limit = {
							faith = scope:holder.faith
						}
						add = 15
					}
					if = { #Or compatible faith at least.
						limit = {
							faith = {
								faith_hostility_level = {
									target = scope:holder.faith
									value = faith_fully_accepted_level
								}
							}
						}
						add = 5
					}
					#if = { #Slightly favor theocracies of same faith.
					#	limit = {
					#		has_government = theocracy_government
					#		faith = scope:holder.faith
					#	}
					#	add = 20 #ex 4
					#}
					if = { #Malus for other governments.
						limit = {
							has_government = republic_government
						}
						subtract = 5
					}
					if = { #Malus for other governments.
						limit = {
							OR = {
								#has_government = tribal_government
								has_government = clan_government
								has_government = mercenary_government
								has_government = holy_order_government
							}
						}
						subtract = 4
					}
					if = { #Slight prioritization for same culture.
						limit = {
							culture = scope:holder.culture
						}
						add = 4
					}
				}
			}
		}
	}

	candidate_score = {
		base = 0
		#Princely Elective uses very similar behavior to Feudal Elective. The few changes to the modifiers are already accounted for in the scripted_modifiers themselves.
		######################	Elector self-voting pattern	##########################
		elector_self_voting_pattern_feudal_elective_modifier = yes

		##########################	Holder voting pattern	##########################
		holder_voting_pattern_feudal_elective_modifier = yes

		##########################	Elector voting patterns (circumstances)	##########################
		elector_voting_pattern_circumstances_feudal_elective_modifier = yes

		##########################	Elector voting patterns (prestige/piety)	##########################
		elector_voting_pattern_prestige_piety_feudal_elective_modifier = yes

		##########################	Elector voting patterns (traits)	##########################
		elector_voting_pattern_traits_feudal_elective_modifier = yes

		elector_voting_pattern_designated_elector_elective_modifier = yes #ARKO
		
		##########################	Elector voting patterns (opinion)	##########################
		elector_voting_pattern_opinion_feudal_elective_modifier = yes
	}
	
	#scope:candidate = Candidate, scope:title = Title, scope:holder_candidate = Candidate currently selected by ruler.
	candidates = {
		add = {
			type = holder_direct_vassals
			limit = {
				feudal_elective_potential_landed_candidate_trigger = yes
				highest_held_title_tier < tier_kingdom
			}
		}
		add = {
			type = title_dejure_vassals
			limit = {
				is_independent_ruler = yes #De jure vassals of the title who are not de facto vassals of the current ruler.
				feudal_elective_potential_landed_candidate_trigger = yes
				highest_held_title_tier < tier_kingdom
			}
		}
		add = {
			type = title_claimants
			limit = {
				feudal_elective_potential_landless_claimant_candidate_trigger = yes
			}
		}
		add = {
			type = holder_close_or_extended_family
			limit = {
				is_close_family_of = scope:holder
				feudal_elective_potential_landless_dynastic_candidate_trigger = yes
			}
		}
	}
}
